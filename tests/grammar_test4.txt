START = Expr
NON_TERMINALS = Expr, Expr', Term, Term', Factor
TERMINALS = ID, NUM, PLUS, MINUS, MUL, DIV, LPAREN, RPAREN

# Grammar Productions
Expr -> Term Expr'
Expr' -> PLUS Expr | MINUS Expr | eps
Term -> Factor Term'
Term' -> MUL Term | DIV Term | eps
Factor -> ID | NUM | LPAREN Expr RPAREN

# Lexical Definitions
ID-> /[a-z]+/
NUM-> /[0-9]+/
PLUS-> /\+/
MINUS-> /-/
MUL -> /\*/
DIV-> /\//
LPAREN-> /\(/
RPAREN-> /\)/
